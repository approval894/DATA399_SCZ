df <- bind_rows(d01_08, d09_14) %>%
  mutate(
    area_km2 = area_m2 / 1e6,
    dist_km  = disttonavriv / 1000,
    is_protected = anp_name != "NONE")
df=df%>%
  filter(year!=2012 & year!=2002)
df %>%
  group_by(year) %>%
  summarise(total_km2 = sum(area_km2, na.rm = TRUE), .groups = "drop") %>%
  ggplot(aes(x = year, y = total_km2)) +
  geom_line() +
  geom_point() +
  labs(
    x = "Year",
    y = "Mining-related deforestation (sq km)",
    title = "Total mining-related deforestation by year"
  )+
  theme_bw()

# total deforestation by year for each district
district_year <- df %>%
  group_by(year, districtcode) %>%
  summarise(total_km2 = sum(area_km2, na.rm = TRUE), .groups = "drop")

ggplot(district_year, aes(x = year, y = total_km2,color=districtcode)) +
  geom_line(linewidth = 0.9) +
  facet_wrap(~ districtcode) +
  labs(
    x = "Year",
    y = "Deforestation (sq km)",
    title = "Mining-related deforestation by district over time",
    color='District Code'
  ) +
  theme_bw()+
  theme(legend.position = "top")

df %>%
  group_by(year, zona_miner) %>%

  filter(threshold == 30) %>%
  pivot_longer(
    cols = starts_with("tc_loss_ha_"),
    names_to = "year",

  mutate(

fig1_data <- mining %>%
  group_by(year) %>%
  summarise(mining_loss_ha = sum(tc_loss_ha, na.rm = TRUE)) %>%
  ungroup()

ggplot(fig1_data, aes(x = year, y = mining_loss_ha)) +
  geom_line(linewidth = 1) +

  scale_y_continuous(labels = comma) +
  labs(
    title = "Global Mining-Related Tree Cover Loss Over Time",
